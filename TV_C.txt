def TV_C(N, seed):
    TV_C = []
    s0 = bin(int(seed))

    for i in range(0, 255):
        bits = ''
        for j in range(0, -(-N//8)):
            bits = s0[2:].rjust(8, '0') + bits
            s0 = bin(int(s0,2) + 1)
        cutoff = abs(len(bits) - N)
        TV_C.insert(i,bits[cutoff:])
    return TV_C
